--Combinación de Tablas: Escribe una consulta que combine las tablas Clientes y
--Cuentas Bancarias para listar todos los clientes con sus respectivas cuentas bancarias.

SELECT 
  c.ID_Cliente,
  c.Nombre,
  o.Numero_Cuenta,
  o.Tipo_Cuenta

FROM ID_Cliente AS c
INNER JOIN Cuentas_Bancarias AS o
ON c.ID_Cliente = o.ID_Cliente

--Uso de Subconsultas: Escribe una subconsulta para listar los clientes que tienen un
--saldo mayor que el saldo promedio de todas las cuentas.
-- Subconsulta para calcular el saldo promedio de todas las cuentas
WITH SaldoPromedio AS (
    SELECT AVG(Saldo) AS Saldo_Promedio
    FROM Cuentas_Bancarias
)

SELECT DISTINCT c.ID_Cliente, c.Nombre
FROM Clientes c
JOIN Cuentas_Bancarias cb ON c.ID_Cliente = cb.ID_Cliente
WHERE cb.Saldo > (SELECT Saldo_Promedio FROM SaldoPromedio)



--Uso de GROUP BY: Crea una consulta que agrupe las transacciones por tipo y calcule el
--número total de transacciones para cada tipo utilizando GROUP BY.

SELECT ID_Tipo_Transacción, COUNT(*) AS Numero_Transacciones
FROM Transacciones
GROUP BY ID_Tipo_Transacción


 
--Uso de HAVING: Escribe una consulta similar a la anterior, pero agrega una condición
--HAVING para mostrar solo los tipos de transacción con más de un cierto número de
--transacciones.

SELECT ID_Tipo_Transacción, COUNT(*) AS Numero_Transacciones
FROM Transacciones
GROUP BY ID_Tipo_Transacción
HAVING COUNT(*) > 3
