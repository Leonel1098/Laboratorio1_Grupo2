create database Laboratorio_1


--Clientes: Esta tabla debe almacenar la información básica de los clientes, incluyendo


create table Clientes

(ID_Cliente int primary key,
Nombre varchar (50),
Dirección varchar (max),
email varchar (max),
Teléfono varchar (50)

--- Cuentas Bancarias: Esta tabla debe contener los detalles de las cuentas bancarias
---asociadas a los clientes, como el número de cuenta, tipo de cuenta (ahorros, corriente),
---saldo, fecha de apertura, y el ID del cliente.

create table Cuentas_Bancarias

(
Numero_Cuenta int primary key,
Tipo_Cuenta varchar (50),
Saldo Decimal (10,2),
Fecha_Apertura Date,
ID_Cliente int)

add constraint chk_Tipo_Cuenta  check (Tipo_Cuenta in ('ahorros','corriente')
CONSTRAINT fk_ID_Cliente FOREIGN KEY (ID_Cliente) REFERENCES Clientes(ID_Cliente),



--- Transacciones: Esta tabla registrará cada transacción financiera realizada, incluyendo el
---tipo de transacción (depósito, retiro, transferencia), la fecha, el monto, la cuenta origen y
---la cuenta destino (si aplica).
---crear tablas de relacion entre transacciones y tipo de transaccion

Create table Transacciones
(ID_Transaccion int primary key,
ID_Tipo_Transacción int,
Fecha_Transaccion date,
Monto	DECIMAL (10,2),
Cuenta_Origen	int ,
Cuenta_Destino int

Create table Tipo_Transaccion
(
ID_Tipo_Transaccion int primary key,
Descripción varchar(50))


--- Empleados: Esta tabla debe incluir la información de los empleados, como nombre, cargo,
---fecha de contratación, salario, y datos de contacto.


CREATE TABLE EMPLEADOS

(
ID_Empleado	int primary key,
Nombre varchar (50),

Fecha_Contratacion date,
